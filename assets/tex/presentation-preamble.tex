% === Custom metropolis + Darmstadt headline + Boadilla footline ===
% Top: miniframes-style section navigation dots (from Darmstadt).
% Bottom: three-box infolines footer with teal gradient (Boadilla layout).
% This keeps metropolis's frametitle and inner theme intact.
\definecolor{ensFootL}{HTML}{D8E3E6}
\definecolor{ensFootM}{HTML}{6B8A93}
\definecolor{ensFootR}{HTML}{23414B}
\makeatletter
\@ifundefined{metroset}{}{%
  \setbeamertemplate{headline}{%
    \begin{beamercolorbox}[wd=\paperwidth,ht=3.7ex,dp=0pt]{section in head/foot}%
      \vbox to 3.7ex{\vskip0pt\insertnavigation{\paperwidth}\vss}%
    \end{beamercolorbox}%
    \vskip1pt%
  }%
  \setbeamercolor{title in head/foot}{bg=ensFootL, fg=black}%
  \setbeamercolor{author in head/foot}{bg=ensFootM, fg=black}%
  \setbeamercolor{date in head/foot}{bg=ensFootR, fg=white}%
  \setbeamertemplate{footline}{%
    \leavevmode%
    \hbox{%
      \begin{beamercolorbox}[wd=.333333\paperwidth,ht=2.5ex,dp=1ex,center]{title in head/foot}%
        \usebeamerfont{title in head/foot}\insertshorttitle%
      \end{beamercolorbox}%
      \begin{beamercolorbox}[wd=.333333\paperwidth,ht=2.5ex,dp=1ex,center]{author in head/foot}%
        \usebeamerfont{author in head/foot}\insertshortauthor%
      \end{beamercolorbox}%
      \begin{beamercolorbox}[wd=.333334\paperwidth,ht=2.5ex,dp=1ex,right]{date in head/foot}%
        \usebeamerfont{date in head/foot}\insertshortdate{}\hspace*{2em}%
        \insertframenumber{} / \inserttotalframenumber\hspace*{2ex}%
      \end{beamercolorbox}%
    }%
    \vskip0pt%
  }%
}
\makeatother
